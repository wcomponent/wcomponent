WComponent

Web Component Library


<element name='wc-login'>
  <template>
    <form data-submit='onSubmit'>
      <label>User Name</label>
      <input data-model='username' type='text' placeholder='User Name'>
      <label>Password</label>
      <input data-model='password' type='password'>
      <button type='submit' class="btn--raised btn--primary">Login</button>
    </form>
  </template>
  <script>
    WComponent('wc-login', {
      callbacks: {
        created: function(){
          console.log('created;login')

        }
      },
      model: {
        username: 'somnath-kokane',
        password: 'wcomponent'
      },
      events: {
        onSubmit: function(){
          alert(JSON.stringify(this.model));
        }
      }
    });
  </script>
</element>
        
          
<element name='wc-admin-login'>
  <template is='wc-template' extends='wc-login'></template>
  <script>
    WComponent('wc-admin-login', {
      'extends': 'wc-login',
      model: {
        username: 'admin',
        password: 'admin'
      }
    });
  </script>
</element>
